147 void operator()(T* p) const {
148 if (p) {
149 if (PAIMON_UNLIKELY(size == 0)) {
150 // This indicates a programming error where the deleter was not properly
151 // initialized with the correct allocation size. In debug builds, this
152 // will trigger an assertion. In release builds, we cannot safely
153 // deallocate without knowing the size, which may lead to undefined behavior.
154 assert(false &&
155 "Mismatched pointer allocation detected - deleter size is zero");
156 return; // Cannot safely deallocate without size information
157 }
158 p->~T();
159 pool.get().Free(reinterpret_cast<void*>(p), size);
160 }
161 }

174 template <typename T, typename... Args>
175 std::unique_ptr<T, AllocatorDelete<T>> AllocateUnique(Args&&... args) {
176 struct DeferCondDeallocate {
177 bool& cond;
178 MemoryPool& pool;
179 void* p;
180 ~DeferCondDeallocate() {
181 if (PAIMON_UNLIKELY(!cond)) {
182 pool.Free(p, sizeof(T));
183 }
184 }
185 };
186 auto p = Malloc(sizeof(T));
187 {
188 bool constructed = false;
189 DeferCondDeallocate handler{constructed, *this, p};
190 new (p) T(std::forward<Args>(args)...);
191 constructed = true;
192 }
193 return std::unique_ptr<T, AllocatorDelete<T>>(reinterpret_cast<T*>(p),
194 AllocatorDelete<T>(*this, sizeof(T)));
195 }
196};
